    // 닫기 외부영역 클릭 처리
    if (isDropdownOpen && !e.target.closest('.mode-dropdown')) setIsDropdownOpen(false);
    if (isMoreMenuOpen && !e.target.closest('.more-menu-container')) setIsMoreMenuOpen(false);

    if (e.target.closest('.no-pin-zone')) return;

    if (uiRef.current) {
      const rect = uiRef.current.getBoundingClientRect();
      const x = ((e.clientX - rect.left) / rect.width) * 100;
      const y = ((e.clientY - rect.top) / rect.height) * 100;

      const newPin = {
        id: Date.now(),
        x,
        y,
        text: "",
        elementContext: e.target.innerText || e.target.tagName || "UI 영역"
      };

      setPins([...pins, newPin]);
      setActivePinId(newPin.id);
      setDraftMemo("");
      setCopied(false);
    }
  };

  const saveMemo = () => {
    if (!draftMemo.trim()) return;
    setPins(pins.map(p => p.id === activePinId ? { ...p, text: draftMemo } : p));
    setActivePinId(null);
    setDraftMemo("");
  };

    setPins(pins.filter(p => p.id !== id));
    if (activePinId === id) setActivePinId(null);

    textArea.select();

        setTimeout(() => setCopied(false), 3000);
